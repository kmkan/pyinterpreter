PROGRAM Part10;
VAR
   number     : INTEGER;
   a, b, c, x : INTEGER;
   y          : REAL;

BEGIN {Part10}
   BEGIN
      number := (+ 1 1);
      a := number;
      b := (+ (* 10 a) (* 10 (DIV number 4)));
      c := (- a (- b))
   END;
   x := (+ 6 5);
   y := (+ 0 3.14);
   { writeln('a = ', a); }
   { writeln('b = ', b); }
   { writeln('c = ', c); }
   { writeln('number = ', number); }
   { writeln('x = ', x); }
   { writeln('y = ', y); }
END.  {Part10}
